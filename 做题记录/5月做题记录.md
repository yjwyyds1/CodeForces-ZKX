```c++
using ull = unsigned long long int;
using ll = long long int;

struct DSU{
    vector<int>f,siz;
    vector<int>d;
    DSU(int n){
        f.resize(n+1);
        d.resize(n+1,0);
        siz.resize(n+1,1);
        iota(f.begin(),f.end(),0);
    }
    int find(int x){
        if(x==f[x])return x;
        int root=find(f[x]);
        d[x]+=d[f[x]];
        return f[x]=root;
    }
    bool merge(int x,int y){
        int rx=find(x);
        int ry=find(y);
        if(rx==ry)return 0;
        f[rx]=ry;
        d[rx]=siz[ry];
        siz[ry]+=siz[rx];
        return 1;
    }   
    bool same(int x,int y){
        return find(x)==find(y);
    }
    ll query(int x,int y){
        if(same(x,y)){
            return abs(d[x]-d[y])-1;
        }
        else{
            return -1;
        }
    }
};
void solve(){
    int q;
    cin>>q;
    int n=30000;
    DSU T(n);
    for(int i=1;i<=q;i++){
        char op;
        cin>>op;
        if(op=='M'){
            int x,y;
            cin>>x>>y;
            T.merge(x,y);
        }
        else{
```

### 代码
```c++
#include <bits/stdc++.h>
const int N=1e5+10;
const int inf = 0x3f3f3f3f;
const int mod = 998244353;
using namespace std;
using ull = unsigned long long int;
using ll = long long int;
using ld = long double;
struct SegmentTree{
    int n;
    vector<vector<ll>> sum,tag;
    SegmentTree(int n_):n(n_),tag(2,vector<ll>(n<<2,1)),sum(2,vector<ll>(n<<2,0)) {}


    void mul(int p,int id,ll k){
        sum[id][p]=(sum[id][p]*k)%mod;
        tag[id][p]=(tag[id][p]*k)%mod;
    }
    void push(int p,int id){
        if(tag[id][p]!=1){
            mul(p<<1,id,tag[id][p]);
            mul(p<<1|1,id,tag[id][p]);
            tag[id][p]=1;
        }
    }
    void add(int p, int id, int l, int r, int x, ll val) {
        if (l == r) {
            sum[id][p] = (sum[id][p] + val) % mod;
            return;
        }
        push(p, id);
        int mid = l + r >> 1;
        if (x <= mid) {
            add(p << 1, id, l, mid, x, val);
        } else {
            add(p << 1 | 1, id, mid + 1, r, x, val);
        }
        sum[id][p] = (sum[id][p << 1] + sum[id][p << 1 | 1]) % mod;
    }
    void RangeMul(int p, int id, int l, int r, int x, int y, ll k) {
        if (y < l || r < x) {
            return;
        }
        if (x <= l && r <= y) {
            mul(p, id, k);
            return;
        }
        push(p, id);
        int mid = l + r >> 1;
        RangeMul(p << 1, id, l, mid, x, y, k);
        RangeMul(p << 1 | 1, id, mid + 1, r, x, y, k);
        sum[id][p] = (sum[id][p << 1] + sum[id][p << 1 | 1]) % mod;
    }
    ll query(int p, int id, int l, int r, int x, int y) {
        if (y < l || r < x) {
            return 0;
        }
        if (x <= l && r <= y) {
            return sum[id][p];
        }
        push(p, id);
        int mid = l + r >>1;
        return (query(p << 1, id, l, mid, x, y) + query(p << 1 | 1, id, mid + 1, r, x, y)) % mod;
    }

    void add(int id,int x,ll val){
        add(1,id,0,n-1,x,val);
    }
    void RangeMul(int id,int x,int y){
        RangeMul(1,id,0,n-1,x,y,2ll);
    }
    ll query(int id,int x,int y){
        return query(1,id,0,n-1,x,y);
    }
};
struct st{
    int l;
    int r;
    int c;
    int idx;
    bool operator < (const st &a)const{
        if(r!=a.r){
            return r<a.r;
        }
        else if(l!=a.l){
            return l<a.l;
        }
        return c<a.c;
    }
};
void solve(){
    int n;
    cin>>n;
    SegmentTree T(n);
    vector<st> s(n+1);
    s[0]={0,0,2};
    for(int i=1;i<=n;i++){
        int l,r,c;
        cin>>l>>r>>c;
        s[i]={l,r,c};
    }
    sort(s.begin(),s.end());
    vector<vector<st>> t(2);
    t[0].push_back({0,0,0,0});
    t[1].push_back({0,0,1,0});
    for(int i=1;i<=n;i++){
       t[s[i].c].push_back({s[i].l,s[i].r,s[i].c,i}); 
    }
    T.add(0,0,1);
    T.add(1,0,1);
    ll ans=1;
    for(int i=1;i<=n;i++){
        int C=s[i].c;
        int I=s[i].l-1;
        int l=0,r=t[C^1].size()-1;
        int pos;
        while(l<=r){
            int mid=l+r>>1;
            if(t[C^1][mid].r<=I){
                pos=mid;
                l=mid+1;
            }
            else{
                r=mid-1;
            }
        }
        ll res=T.query(C^1,0,t[C^1][pos].idx);
        T.add(C,i,res);
        T.RangeMul(C^1,0,t[C^1][pos].idx);
        ans=(res+ans)%mod;
    }
    cout<<ans<<'\n';
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
```

```c++
ll qmi(ll a,ll b){
    ll res=1;

            }
        }
    }

}
```

```c++
ll qmi(ll a,ll b){
    ll res=1;
    while(b){
        if(b&1)res=res*a%mod;
        a=a*a%mod;
        b>>=1;
    }
    return res%mod;
}
void solve(){
    int n;
    cin>>n;
    ll a[n+1],b[n+1],p[n+1];
    p[0]=0;
    for(int i=1;i<=n;i++){
        cin>>a[i];

        p[i]=0;
    }
    stack<PII> s;
    ll ans=0;
    for(int i=1;i<=n;i++){
        ll cnt=0;
        while(b[i]%2==0)b[i]/=2,cnt++;
        p[i]=p[i-1]+cnt;
        while(!s.empty()&&log2(a[i])+p[i-1]>=log2(a[s.top().first])+p[s.top().first-1]){
            auto [x,y]=s.top();

            ans=(ans-(b[x]*(((qmi(2,y)-1)+mod)%mod))%mod+mod)%mod;
            cnt+=y;
        }
        s.push({i,cnt});
        ans=(ans+(qmi(2,cnt)*b[i])%mod)%mod;
        cout<<ans<<" ";
    }

}
```

```c++
using ll = long long int;
using ld = long double;

void dfs(int x,int f){
    for(auto [y,w]:e[x]){
        if(y==f)continue;
        else{
            if(!v[y]){
                ans+=w;
                dfs(y,x);
            }
        }   
    }

void solve(){
    int n,m;
    cin>>n>>m;
    int Q;
    vector<int> lx(n+1,0);
    for(int i=1;i<=m;i++){
        int x;

        Q=x;
    }

    for(int i=1;i<n;i++){
        int u,v,w;
        cin>>u>>v>>w;
        e[u].push_back({v,w});
        e[v].push_back({u,w});
        d[u]++,d[v]++;
    }
    queue<int> q;
    for(int i=1;i<=n;i++){

            if(d[y]==1&&!lx[y]){
                q.push(y);
            }
        }
    }
```
